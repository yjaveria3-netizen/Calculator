// Global state for memory and history
std::vector<double> history;
std::vector<std::string> historyLabels;
double memory = 0.0;
const int MAX_HISTORY = 50;

// Add calculation to history (FIFO with limit)
void addToHistory(double value, const std::string &label = "") {
    history.push_back(value);
    historyLabels.push_back(label);
    
    // Keep only last 50 entries
    if (history.size() > MAX_HISTORY) {
        history.erase(history.begin());
        historyLabels.erase(historyLabels.begin());
    }
}

// Memory operations (MS, MR, MC, M+)

4. Expression Parser (Shunting Yard Algorithm)

The crown jewel: an expression parser that handles complex mathematical expressions with proper operator precedence and parentheses.

Expression Evaluation Engine C++ 
// Operator precedence levels (higher = evaluated first)
int getPrecedence(char op) {
    if (op == '+' || op == '-') return 1;
    if (op == '*' || op == '/') return 2;
    if (op == '^') return 3;
    return 0;
}

// Apply binary operation
double applyOperation(double a, double b, char op) {
    switch (op) {
        case '+': return a + b;
        case '-': return a - b;
        case '*': return a * b;
        case '/':
            if (b == 0) throw std::runtime_error("Division by zero");
            return a / b;
        case '^': return std::pow(a, b);
        default: return 0;
    }
}

// Main expression evaluator using two-stack algorithm
double evaluateExpression(const std::string &expr) {
    std::stack<double> values;  // Stack for numbers
    std::stack<char> ops;      // Stack for operators
    
    for (int i = 0; i < expr.length(); i++) {
        // Skip whitespace
        if (isspace(expr[i])) continue;
        
        // Parse numbers (including decimals)
        if (isdigit(expr[i]) || expr[i] == '.') {
            std::string numStr;
            while (i < expr.length() && 
                   (isdigit(expr[i]) || expr[i] == '.')) {
                numStr += expr[i++];
            }
            i--;
            values.push(std::stod(numStr));
        }
        // Handle opening parenthesis
        else if (expr[i] == '(') {
            ops.push(expr[i]);
        }
        // Handle closing parenthesis
        else if (expr[i] == ')') {
            while (!ops.empty() && ops.top() != '(') {
                double b = values.top(); values.pop();
                double a = values.top(); values.pop();
                char op = ops.top(); ops.pop();
                values.push(applyOperation(a, b, op));
            }
            if (!ops.empty()) ops.pop(); // Remove '('
        }
        // Handle operators
        else if (expr[i] == '+' || expr[i] == '-' || 
                 expr[i] == '*' || expr[i] == '/' || expr[i] == '^') {
            // Process operators with higher/equal precedence
            while (!ops.empty() && 
                   getPrecedence(ops.top()) >= getPrecedence(expr[i])) {
                double b = values.top(); values.pop();
                double a = values.top(); values.pop();
                char op = ops.top(); ops.pop();
                values.push(applyOperation(a, b, op));
            }
            ops.push(expr[i]);
        }
    }
    
    // Process remaining operators
    while (!ops.empty()) {
        double b = values.top(); values.pop();
        double a = values.top(); values.pop();
        char op = ops.top(); ops.pop();
        values.push(applyOperation(a, b, op));
    }
    
    return values.top();
}

Matrix Multiplication Algorithm C++ 
void matrixMultiplication() {
    int r1, c1, r2, c2;
    
    // Get dimensions for Matrix A
    std::cout << "Matrix A - Rows: ";
    std::cin >> r1;
    std::cout << "Matrix A - Columns: ";
    std::cin >> c1;
    
    // Get dimensions for Matrix B
    std::cout << "Matrix B - Rows: ";
    std::cin >> r2;
    std::cout << "Matrix B - Columns: ";
    std::cin >> c2;
    
    // Validate: A columns must equal B rows
    if (c1 != r2) {
        std::cout << theme->error 
                  << "Error: Cannot multiply! A columns must equal B rows."
                  << theme->reset << std::endl;
        return;
    }
    
    // Initialize matrices
    std::vector<std::vector<double>> A(r1, std::vector<double>(c1));
    std::vector<std::vector<double>> B(r2, std::vector<double>(c2));
    std::vector<std::vector<double>> result(r1, std::vector<double>(c2, 0));
    
    // Input Matrix A
    std::cout << "Enter elements for Matrix A:\n";
    for (int i = 0; i < r1; i++)
        for (int j = 0; j < c1; j++)
            std::cin >> A[i][j];
    
    // Input Matrix B
    std::cout << "Enter elements for Matrix B:\n";
    for (int i = 0; i < r2; i++)
        for (int j = 0; j < c2; j++)
            std::cin >> B[i][j];
    
    // Perform multiplication: C[i][j] = Î£(A[i][k] * B[k][j])
    for (int i = 0; i < r1; i++) {
        for (int j = 0; j < c2; j++) {
            for (int k = 0; k < c1; k++) {
                result[i][j] += A[i][k] * B[k][j];
            }
        }
    }
    
    // Display result
    std::cout << theme->success << "\nResult Matrix:\n";
    for (int i = 0; i < r1; i++) {
        for (int j = 0; j < c2; j++) {
            std::cout << std::setw(8) << result[i][j] << " ";
        }
        std::cout << "\n";
    }
    std::cout << theme->reset;
}

Euclidean Algorithm (GCD)

Purpose: Find greatest common divisor efficiently

Complexity: O(log min(a,b))

Method: Recursive remainder calculation until remainder is 0

int gcd(int a, int b) {
    if (b == 0) return a;
    return gcd(b, a % b);

Example: Well-Documented Function C++ 
/**
 * Calculates the factorial of a number using iteration
 * 
 * @param n The number to calculate factorial for (must be non-negative)
 * @return The factorial value, or 0 if input is invalid
 * 
 * Note: Maximum supported value is 170 due to double precision limits
 * Example: factorial(5) returns 120.0
 */
double factorial(int n) {
    // Input validation
    if (n < 0) {
        std::cout << theme->error 
                  << "Error: Factorial undefined for negative numbers!" 
                  << theme->reset << std::endl;
        return 0;
    }
    
    // Handle overflow prevention
    if (n > 170) {
        std::cout << theme->warning 
                  << "Warning: Result will overflow. Max supported is 170!" 
                  << theme->reset << std::endl;
    }
    
    // Base case
    if (n == 0 || n == 1) return 1.0;
    
    // Iterative calculation (more efficient than recursion)
    double result = 1.0;
    for (int i = 2; i <= n; i++) {
        result *= i;
    }
    
    return result;
}
